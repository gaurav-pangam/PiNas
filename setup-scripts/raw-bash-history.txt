        }
        echo "$(date): Mount failed, retrying..." >> "$LOGFILE"
    else
        echo "$(date): Attempt $i - Device not found, retrying..." >> "$LOGFILE"
    fi
    sleep "$DELAY"
done

echo "$(date): Failed to mount after $RETRIES attempts." >> "$LOGFILE"
exit 1
